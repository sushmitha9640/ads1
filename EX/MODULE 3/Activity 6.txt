public class Sort
{
	public static void main(String args[])
	{
	int a[] = {5,8,9,3,6,4,2,1,0,5,6,7,8,9};
	int count=0;
	for(int i=0;i<a.length;i++)
	{
		for(int j = i+1;j<a.length;j++)
		{
			if(a[i]>a[j])
			{
				int temp;
				temp=a[i];
				a[i]=a[j];
				a[j]=temp;
			}
			
		}
		System.out.print(a[i]);
	}
		for(int i=0;i<a.length-1;i++)
		{
		
			if(a[i]==a[i+1])
			{
				count++;
			}
			
		}
		System.out.println();
		System.out.println(count);
	}
	
}
